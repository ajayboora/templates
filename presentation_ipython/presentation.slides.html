from pylab import *

x = linspace(0, 1)

figure()
for power in [1, 2, 3, 4, 5]:
    plot(x, x**power, label='x**{0}'.format(power))
legend(loc='best')
xlabel('x')
ylabel('x**power')
title('Different powers')
show()
